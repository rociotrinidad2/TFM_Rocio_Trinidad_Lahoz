# FunciÃ³n para limpiar y convertir datos numÃ©ricos
clean_numeric_data <- function(data) {
  # Reemplazar cualquier carÃ¡cter no numÃ©rico (por ejemplo, comas) por un punto
  data_clean <- gsub(",", ".", data)  # Cambiar comas por puntos decimales, si es necesario
  data_clean <- as.numeric(data_clean)  # Convertir a numÃ©rico
  return(data_clean)
}

# Crear la funciÃ³n para realizar el test de Shapiro-Wilk con verificaciÃ³n del tamaÃ±o de la muestra
shapiro_results <- function(data, var, group_name) {
  if (var %in% colnames(data)) {
    # Eliminar valores NA y contar las observaciones vÃ¡lidas
    valid_data <- na.omit(data[[var]])
    n <- length(valid_data)
    
    if (n >= 3) {
      # Limpiar y convertir los datos a numÃ©rico
      valid_data <- clean_numeric_data(valid_data)
      
      # Verificar si la variable es numÃ©rica despuÃ©s de la limpieza
      if (is.numeric(valid_data)) {
        # Realizar el test de Shapiro-Wilk si el tamaÃ±o de la muestra es adecuado
        shapiro_test <- shapiro.test(valid_data)
        
        # Agregar la interpretaciÃ³n en funciÃ³n del valor p
        interpretation <- ifelse(shapiro_test$p.value < 0.05, "No es normal", "Es normal")
        
        return(data.frame(
          Grupo = group_name,
          Variable = var,
          W = shapiro_test$statistic,
          p_value = shapiro_test$p.value,
          Interpretacion = interpretation
        ))
      } else {
        # Si la variable no es numÃ©rica, devolver NA
        return(data.frame(
          Grupo = group_name,
          Variable = var,
          W = NA,
          p_value = NA,
          Interpretacion = "No es numÃ©rica"
        ))
      }
    } else {
      # Si el tamaÃ±o de la muestra es menor a 3, devolver NA y un mensaje indicando el problema
      warning(paste("Advertencia: La variable", var, "en el grupo", group_name, "tiene menos de 3 observaciones vÃ¡lidas."))
      return(data.frame(
        Grupo = group_name,
        Variable = var,
        W = NA,
        p_value = NA,
        Interpretacion = "TamaÃ±o de muestra insuficiente"
      ))
    }
  } else {
    # Si la variable no existe, devolver NA
    return(data.frame(
      Grupo = group_name,
      Variable = var,
      W = NA,
      p_value = NA,
      Interpretacion = NA
    ))
  }
}

# Crear la funciÃ³n para realizar la prueba de Levene
levene_results <- function(data1, data2, var) {
  if (var %in% colnames(data1) && var %in% colnames(data2)) {
    # Eliminar NA y convertir a numÃ©rico si es necesario
    valid_data1 <- as.numeric(na.omit(data1[[var]]))  # Asegurarse de que la columna es numÃ©rica
    valid_data2 <- as.numeric(na.omit(data2[[var]]))  # Asegurarse de que la columna es numÃ©rica
    
    # Verificar si las columnas ahora son numÃ©ricas
    if (is.numeric(valid_data1) && is.numeric(valid_data2)) {
      # Combinar ambos grupos
      combined_data <- data.frame(
        Grupo = factor(c(rep("PGT", length(valid_data1)), rep("No PGT", length(valid_data2)))),  # Asegurarse de que 'Grupo' sea un factor
        Variable = c(valid_data1, valid_data2)
      )
      
      # Realizar la prueba de Levene (con la mediana como centro)
      levene_test <- leveneTest(Variable ~ Grupo, data = combined_data, center = median)
      
      # Determinar la interpretaciÃ³n de la prueba de Levene
      p_value <- levene_test$`Pr(>F)`[1]  # Acceder al primer valor del p-valor
      if (p_value < 0.05) {
        interpretacion <- "No homogÃ©neas"
      } else {
        interpretacion <- "HomogÃ©neas"
      }
      
      return(data.frame(
        Variable = var,
        F = levene_test$`F value`[1],  # Acceder al primer valor de F
        p_value = p_value,
        Interpretacion = interpretacion
      ))
    } else {
      # Si alguna de las variables no es numÃ©rica, devolver NA
      return(data.frame(
        Variable = var,
        F = NA,
        p_value = NA,
        Interpretacion = "No es numÃ©rica"
      ))
    }
  } else {
    # Si la variable no existe en ambos datasets, devolver NA
    return(data.frame(
      Variable = var,
      F = NA,
      p_value = NA,
      Interpretacion = NA
    ))
  }
}

  # Filtrar filas completas (sin NA) para las dos columnas
  datos <- na.omit(pacientes_37[, c(variable, "Ciclo con especificacion PGT")])
  
  # Convertir a factores (si no lo son)
  datos[[variable]] <- as.factor(datos[[variable]])
  datos[["Ciclo con especificacion PGT"]] <- as.factor(datos[["Ciclo con especificacion PGT"]])
  
  # Crear la tabla de contingencia
  tabla <- table(datos[[variable]], datos[["Ciclo con especificacion PGT"]])
  
  # Inicializar variables para los resultados
  chi2 <- NA
  p_value <- NA
  metodo <- NA
  interpretacion <- NA

  
  # Realizar el test si la tabla no estÃ¡ vacÃ­a
  if (length(tabla) > 0 && all(dim(tabla) > 1)) {
    tryCatch({
      if (all(tabla >= 5)) {
        # Test de Chi-cuadrado
        resultado <- chisq.test(tabla)
        metodo <- "Chi-cuadrado"
        chi2 <- resultado$statistic
        p_value <- resultado$p.value
      } else {
        # Test exacto de Fisher
        resultado <- fisher.test(tabla)
        metodo <- "Fisher exacto"
        p_value <- resultado$p.value
      }
      # InterpretaciÃ³n basada en el p-valor
      interpretacion <- if (!is.na(p_value) && p_value < 0.05) "Significativo" else "No significativo"
    }, error = function(e) {
      metodo <- "Error en el test"
    })
  } else {
    metodo <- "Tabla vacÃ­a o dimensiones insuficientes"
  }
  
  # Almacenar el resultado en el data frame
  resultados_chi2 <- rbind(resultados_chi2, data.frame(
    Variable = variable,
    Chi2 = ifelse(metodo == "Fisher exacto", NA, chi2),
    p_value = p_value,
    Metodo = metodo,
    Interpretacion = interpretacion
  ))
